Daily Auth Flow — All Brokers (runs at 8 AM via cron)
"""
DAILY AUTH SCHEDULER (8 AM cron job)
Renews all broker sessions automatically — user never logs in manually
"""
import asyncio, pyotp, requests

class HormuzAuth:
    def __init__(self, vault_creds):  # Decrypted from Hormuz vault
        self.v = vault_creds

    async def renew_all(self):
        results = await asyncio.gather(
            self.zerodha(),
            self.angel_one(),
            self.groww(),
            return_exceptions=True
        )
        for r in results:
            print(r if isinstance(r, Exception) else f"✓ {r['broker']} session renewed")

    async def zerodha(self):
        # Uses Playwright/Selenium to handle the login redirect
        # Captures request_token → exchanges for access_token
        totp = pyotp.TOTP(self.v["zerodha"]["totp"]).now()
        # ... browser automation to get request_token ...
        # kite.generate_session(request_token, api_secret)
        return {"broker": "Zerodha", "expires": "6AM tomorrow"}

    async def angel_one(self):
        # Angel has direct API login — no browser needed!
        totp = pyotp.TOTP(self.v["angel"]["totp"]).now()
        r = requests.post(
            "https://apiconnect.angelbroking.com/rest/auth/.../loginByPassword",
            json={"clientcode": self.v["angel"]["client_id"],
                  "password": self.v["angel"]["mpin"], "totp": totp},
            headers={"X-PrivateKey": self.v["angel"]["api_key"]}
        )
        jwt = r.json()["data"]["jwtToken"]
        return {"broker": "Angel One", "token": jwt, "expires": "+24hrs"}

    async def groww(self):
        # OAuth2: use refresh_token (30-day validity) to get new access_token
        r = requests.post("https://groww.in/v1/oauth/token", data={
            "grant_type": "refresh_token",
            "refresh_token": self.v["groww"]["refresh_token"],
            "client_id": self.v["groww"]["client_id"]
        })
        return {"broker": "Groww", "token": r.json()["access_token"], "expires": "+24hrs"}
